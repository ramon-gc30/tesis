# variables continuas
calcular_descriptivos <- function(datos, variable){
  # se enmascara nombre de columna
  nombre_var <- as_label(enquo(variable))
  
  datos |> 
    filter(!is.na({{  variable  }})) |> 
    summarise(
      # se inserta nombre de la columna
      variable = nombre_var, 
      minimo = min({{  variable  }}),
      promedio = survey_mean({{  variable  }}, vartype = NULL),
      desv_est = survey_sd({{  variable  }}),
      maximo = max({{  variable  }})
    )
}

# variables continuas
calcular_descriptivos <- function(datos, variable){

  poblacion <- get(datos)
  variable_sym <- sym(variable)
  
  datos |> 
    filter(!is.na(  !!variable_sym  )) |> 
    summarise(
      datos = datos,
      variable = variable, 
      min = survey_quantile(  
        !!variable_sym, 
        quantiles = 0, 
        vartype = NULL
      ),
      media = survey_mean(  !!variable_sym  , vartype = NULL),
      mediana = survey_median(  !!variable_sym  , varype = NULL),
      var = survey_var(  !!variable_sym  , vartype = NULL),
      sd = survey_sd(  !!variable_sym  ),
      max = survey_quantile(  
        !!variable_sym, 
        quantiles = 1, 
        vartype = NULL
      )
    )
}
